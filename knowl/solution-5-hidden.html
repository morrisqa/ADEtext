\begin{equation*} z(t)=z(0)-\int_0^t |z(s)| \; ds = \int_0^t |z(s)| \; ds \end{equation*}

and

\begin{equation*} w(t)=w(0)-\int_0^t |w(s)| \; ds = \int_0^t |w(s)| \; ds, \end{equation*}

respectively, for all \(t \in \mathbb{R}\text{.}\) Then,

\begin{equation*} |w(t)-z(t)|= \left| \int_0^t |w(s)| \; ds - \int_0^t |z(s)| \; ds \right| = \left| \int_0^t |w(s)| - |z(s)| \; ds \right| \; \; \; \forall t \in \mathbb{R}. \end{equation*}
